/* Automatically logs in Struber Creative employees when thy navigate to wp-admin */
if ( $GLOBALS['pagenow'] === 'wp-login.php' && $_SERVER['REMOTE_ADDR'] == 'YOUR IP ADDRESS') {
// We're on the login page!
// Automatic login //
$user = get_user_by('ID', 1 );

// Redirect URL //
if ( !is_wp_error( $user ) )
{
wp_clear_auth_cookie();
wp_set_current_user ( $user->ID );
wp_set_auth_cookie ( $user->ID );

$redirect_to = user_admin_url();
wp_safe_redirect( $redirect_to );
exit();
}
}
